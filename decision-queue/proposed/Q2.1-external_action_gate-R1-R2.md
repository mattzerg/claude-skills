PR_NUM_RE = re.compile(r"\bpr\s+(?:merge|close)\s+(\d{1,5})\b")
SELF_LOGIN = "mattzerg"


def _pr_review_decision(pr: str) -> str | None:
    """Query gh for the current review decision on PR #pr (e.g. APPROVED)."""
    try:
        r = subprocess.run(
            ["gh", "pr", "view", pr, "--repo", "Epoch-ML/zerg",
             "--json", "reviewDecision"],
            capture_output=True, text=True, timeout=5,
        )
        if r.returncode != 0:
            return None
        d = json.loads(r.stdout)
        return d.get("reviewDecision")
    except Exception:
        return None


def _pr_author(pr: str) -> str | None:
    try:
        r = subprocess.run(
            ["gh", "pr", "view", pr, "--repo", "Epoch-ML/zerg",
             "--json", "author"],
            capture_output=True, text=True, timeout=5,
        )
        if r.returncode != 0:
            return None
        d = json.loads(r.stdout)
        return (d.get("author") or {}).get("login")
    except Exception:
        return None


def _auto_allow_pr_action(slug: str, cmd: str) -> tuple[bool, str]:
    """Q2.1 R1+R2 — auto-allow common false-positive bypass classes.

    R1: gh pr merge on a PR with reviewDecision=APPROVED.
    R2: gh pr close on a PR authored by self (SELF_LOGIN).
    Returns (allow, reason_for_telemetry).
    """
    if slug not in ("gh-pr-merge", "gh-pr-close"):
        return False, ""
    m = PR_NUM_RE.search(cmd)
    if not m:
        return False, ""
    pr_num = m.group(1)
    if slug == "gh-pr-merge":
        decision = _pr_review_decision(pr_num)
        if decision == "APPROVED":
            return True, f"auto-allow:R1 PR#{pr_num} reviewDecision=APPROVED"
    elif slug == "gh-pr-close":
        author = _pr_author(pr_num)
        if author == SELF_LOGIN:
            return True, f"auto-allow:R2 PR#{pr_num} author=self ({SELF_LOGIN})"
    return False, ""

def _classify(cmd: str) -> tuple[str, str] | None:
